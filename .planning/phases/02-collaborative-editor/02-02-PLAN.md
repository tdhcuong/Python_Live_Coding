**Step 2: Create `frontend/src/editor/provider.js` — RoomProvider class (per D-04, Research Pattern 1):**

```javascript
import * as Y from 'yjs'
import { encodeAwarenessUpdate, applyAwarenessUpdate } from 'y-protocols/awareness'

export const YTEXT_KEY = 'python-code'  // Single source of truth (Pitfall 5)

export class RoomProvider {
  constructor(ydoc, awareness, wsSend) {
    this._ydoc = ydoc
    this.awareness = awareness
    this._wsSend = wsSend

    this._updateHandler = (update, origin) => {
      if (origin !== this) {
        const b64 = btoa(String.fromCharCode(...update))
        wsSend({ type: 'yjs_update', update: b64 })
      }
    }
    ydoc.on('update', this._updateHandler)

    this._awarenessHandler = ({ changed }) => {
      const update = encodeAwarenessUpdate(awareness, [awareness.clientID])
      const b64 = btoa(String.fromCharCode(...update))
      wsSend({ type: 'awareness_update', update: b64 })
    }
    awareness.on('change', this._awarenessHandler)
  }

  applyRemoteUpdate(b64) {
    const bytes = Uint8Array.from(atob(b64), c => c.charCodeAt(0))
    Y.applyUpdate(this._ydoc, bytes, this)
  }

  applyRemoteAwareness(b64) {
    const bytes = Uint8Array.from(atob(b64), c => c.charCodeAt(0))
    applyAwarenessUpdate(this.awareness, bytes, this)
  }

  destroy() {
    this._ydoc.off('update', this._updateHandler)
    this.awareness.off('change', this._awarenessHandler)
  }
}
```

Key points:
- `origin !== this` prevents echo loop (Pitfall 3)
- `btoa(String.fromCharCode(...update))` is the correct Uint8Array-to-base64 pattern (Pitfall 2)
- `YTEXT_KEY` exported as constant to prevent key mismatch (Pitfall 5)
- `encodeAwarenessUpdate` and `applyAwarenessUpdate` imported from `y-protocols/awareness`

```javascript
import { yCollab } from 'y-codemirror.next'
import * as Y from 'yjs'

export function createEditor(container, ytext, awareness, myUser) {
  const undoManager = new Y.UndoManager(ytext)

  // D-07: Set awareness user info — drives cursor color/label in remote views
  awareness.setLocalStateField('user', {
    name: myUser.name,
    color: myUser.color,
    colorLight: myUser.color + '40',  // 25% opacity for selection highlight
  })

  const state = EditorState.create({
    doc: ytext.toString(),
    extensions: [
      lineNumbers(),                                    // EDIT-04: line numbers
      oneDark,                                          // Dark theme matching gray-900 palette
      python(),                                         // EDIT-02: Python syntax highlighting
      yCollab(ytext, awareness, { undoManager }),        // EDIT-01/EDIT-03: CRDT sync + remote cursors
      EditorView.theme({ '&': { height: '100%' } }),    // Pitfall 6: explicit height
    ],
  })

  return new EditorView({ state, parent: container })
}
```

```javascript

  if (provider) {
    provider.applyRemoteUpdate(msg.update)

},
```
